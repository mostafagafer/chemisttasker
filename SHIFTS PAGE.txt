const generateMockPosts = () => {
  const LOCATION_STRUCTURE = {
    'QLD': ['North Brisbane', 'South Brisbane', 'Gold Coast Region', 'Sunshine Coast', 'Townsville', 'Cairns', 'Outback QLD'],
    'NSW': ['Sydney CBD', 'Western Sydney', 'North Shore', 'Newcastle', 'Central Coast', 'Byron Bay', 'Remote NSW'],
    'VIC': ['Melbourne CBD', 'St Kilda', 'Geelong', 'Mornington Peninsula'],
    'WA': ['Perth', 'Fremantle', 'Broome', 'Regional WA'],
    'SA': ['Adelaide', 'Glenelg', 'Barossa Valley']
  };

  const pharmacies = ['Chemist Warehouse', 'Priceline Pharmacy', 'Blooms The Chemist', 'TerryWhite Chemmart', 'Local Care Pharmacy', 'Capital Chemist'];
  const roles = ['Pharmacist', 'Dispensary Tech', 'Pharmacy Assistant', 'Locum Pharmacist', 'Student Intern'];
  
  const stateKeys = Object.keys(LOCATION_STRUCTURE);

  const randomPosts = Array.from({ length: 60 }, (_, i) => {
    const state = stateKeys[Math.floor(Math.random() * stateKeys.length)];
    const regionsInState = LOCATION_STRUCTURE[state];
    const city = regionsInState[Math.floor(Math.random() * regionsInState.length)];

    const pharmacy = pharmacies[Math.floor(Math.random() * pharmacies.length)];
    const role = roles[Math.floor(Math.random() * roles.length)];
    
    let baseRate = 30;
    if (role.includes('Pharmacist')) baseRate = 65;
    if (role.includes('Tech')) baseRate = 35;
    
    // Shift Generation
    const rand = Math.random();
    let numShifts = 1;
    if (rand > 0.92) numShifts = Math.floor(Math.random() * 6) + 5; 
    else if (rand > 0.75) numShifts = Math.floor(Math.random() * 3) + 2;

    const shifts = [];
    const startDate = new Date();
    startDate.setDate(startDate.getDate() + Math.floor(Math.random() * 30));

    for (let j = 0; j < numShifts; j++) {
      const currentShiftDate = new Date(startDate);
      currentShiftDate.setDate(startDate.getDate() + j);
      const isWeekend = currentShiftDate.getDay() === 0 || currentShiftDate.getDay() === 6;
      const shiftRate = isWeekend ? Math.floor(baseRate * 1.2) : Math.floor(Math.random() * (baseRate + 10 - baseRate) + baseRate);
      const startHour = Math.floor(Math.random() * (14 - 8) + 8); 
      const duration = 8;
      
      shifts.push({
        id: `${i}-${j}`, 
        date: currentShiftDate.toISOString().split('T')[0],
        startTime: `${startHour.toString().padStart(2, '0')}:00`,
        endTime: `${(startHour + duration).toString().padStart(2, '0')}:00`,
        rate: shiftRate,
        startHour, 
      });
    }

    const maxRate = Math.max(...shifts.map(s => s.rate));
    const isUrgent = maxRate > baseRate + 20;
    
    // Increased probabilities for demo purposes
    const isNegotiable = role === 'Locum Pharmacist' && Math.random() > 0.4;
    const isFlexible = Math.random() > 0.5; 
    const isRural = city.includes('Remote') || city.includes('Outback') || city.includes('Regional');
    const hasTravel = isRural || Math.random() > 0.85;
    const hasAccommodation = isRural || Math.random() > 0.9;
    
    // High chance of allowing partial applications for multi-shift posts
    const allowPartial = numShifts > 1 && Math.random() > 0.2;

    return {
      id: i + 1,
      pharmacy,
      city,   
      state,  
      address: `${Math.floor(Math.random() * 100) + 1} Main St, ${city}, ${state}`,
      role,
      isUrgent,
      distance: Math.floor(Math.random() * 20) + 1,
      isNegotiable,
      isFlexible,
      hasTravel,
      hasAccommodation,
      allowPartial, 
      shifts, 
      description: "We are looking for a reliable locum to help with our busy dispensary. Dispense software is Fred. Great team environment."
    };
  });

  // --- DEMO OVERRIDE: GUARANTEE FIRST POST HAS ALL FEATURES ---
  // This ensures you see an example of "Counter Offer per Shift" immediately
  const demoPost = {
    ...randomPosts[0],
    id: 'demo-perfect-post',
    pharmacy: 'Chemist Tasker Demo Pharmacy',
    role: 'Locum Pharmacist',
    city: 'Gold Coast Region',
    state: 'QLD',
    isNegotiable: true,
    isFlexible: true,
    allowPartial: true, // Key for partial apply
    isUrgent: true,
    shifts: [
      { id: 'd1', date: new Date().toISOString().split('T')[0], startTime: '09:00', endTime: '17:00', rate: 75 },
      { id: 'd2', date: new Date(Date.now() + 86400000).toISOString().split('T')[0], startTime: '09:00', endTime: '17:00', rate: 75 },
      { id: 'd3', date: new Date(Date.now() + 172800000).toISOString().split('T')[0], startTime: '09:00', endTime: '13:00', rate: 85 } // Weekend rate
    ],
    description: "**DEMO POST**: This post allows partial application. Expand details and check specific shifts to see the 'Counter Selected' button appear."
  };

  return [demoPost, ...randomPosts.slice(1)];
};

const IndeterminateCheckbox = ({ checked, indeterminate, onChange, className }) => {
  const ref = useRef(null);
  useEffect(() => {
    if (ref.current) {
      ref.current.indeterminate = indeterminate;
    }
  }, [indeterminate]);
  return <input type="checkbox" ref={ref} checked={checked} onChange={onChange} className={className} />;
};

const CounterOfferModal = ({ post, selectedShiftIds, onClose, onSubmit }) => {
  const isLocumPharmacist = post.role === 'Locum Pharmacist';
  const canNegotiateRate = post.isNegotiable && isLocumPharmacist;
  const canNegotiateTime = post.isFlexible;

  // Filter shifts based on selection if provided (partial apply/counter)
  const shiftsToNegotiate = useMemo(() => {
    if (selectedShiftIds && selectedShiftIds.size > 0) {
      return post.shifts.filter(s => selectedShiftIds.has(s.id));
    }
    return post.shifts;
  }, [post.shifts, selectedShiftIds]);

  const [formData, setFormData] = useState({
    requestTravel: false,
    shifts: shiftsToNegotiate.map(s => ({
      id: s.id,
      date: s.date, // keep date for display
      startTime: s.startTime,
      endTime: s.endTime,
      rate: s.rate
    }))
  });

  const handleShiftChange = (shiftId, field, value) => {
    setFormData(prev => ({
      ...prev,
      shifts: prev.shifts.map(s => s.id === shiftId ? { ...s, [field]: value } : s)
    }));
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    onSubmit(formData);
  };

  return (
    <div className="fixed inset-0 z-[100] flex items-center justify-center p-4 bg-black/60 backdrop-blur-sm animate-in fade-in duration-200">
      <div className="bg-white rounded-xl shadow-2xl w-full max-w-2xl max-h-[90vh] overflow-hidden flex flex-col animate-in zoom-in-95 duration-200">
        <div className="p-5 border-b border-gray-100 flex justify-between items-center bg-slate-50">
          <div>
            <h3 className="font-bold text-lg text-slate-800">Submit Counter Offer</h3>
            <p className="text-sm text-slate-500">
              {post.pharmacy} â€¢ {post.role} 
              {selectedShiftIds?.size > 0 && <span className="ml-2 bg-blue-100 text-blue-700 px-2 py-0.5 rounded text-xs font-bold">{selectedShiftIds.size} Shifts Selected</span>}
            </p>
          </div>
          <button onClick={onClose} className="p-2 hover:bg-slate-200 rounded-full transition-colors">
            <X size={20} className="text-slate-500" />
          </button>
        </div>
        <div className="overflow-y-auto p-6 space-y-6 flex-1">
          <div className="bg-blue-50 border border-blue-100 rounded-lg p-3 flex gap-3 text-sm text-blue-800">
            <AlertCircle size={20} className="shrink-0" />
            <div>
              <p>
                <strong>Negotiation Options:</strong>
                {canNegotiateRate 
                  ? " You may propose a new rate." 
                  : " Rate is fixed per award/agreement for this role."}
                {canNegotiateTime 
                  ? " Shift times are flexible." 
                  : " Shift times are fixed."}
              </p>
            </div>
          </div>
          <form id="counter-form" onSubmit={handleSubmit} className="space-y-6">
            <div>
              <h4 className="font-bold text-sm text-slate-700 uppercase tracking-wide mb-3 flex items-center gap-2">
                <Clock size={16} /> Propose Shift Details
              </h4>
              <div className="space-y-3">
                {formData.shifts.map((shift) => {
                  return (
                    <div key={shift.id} className="bg-white border border-gray-200 rounded-lg p-4 shadow-sm">
                      <div className="flex flex-col sm:flex-row sm:items-center justify-between gap-4 mb-3 pb-3 border-b border-gray-50">
                        <span className="font-semibold text-slate-700 flex items-center gap-2">
                          <Calendar size={16} className="text-slate-400" />
                          {new Date(shift.date).toLocaleDateString('en-AU', { weekday: 'short', day: 'numeric', month: 'short' })}
                        </span>
                        {!isLocumPharmacist && (
                          <span className="text-xs font-bold text-slate-500 bg-slate-100 px-2 py-1 rounded">
                            Award Rate Applies
                          </span>
                        )}
                      </div>
                      <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
                        <div className="space-y-1">
                          <label className="text-xs font-semibold text-slate-500">Time</label>
                          {canNegotiateTime ? (
                            <div className="flex items-center gap-2">
                              <input 
                                type="time" 
                                value={shift.startTime}
                                onChange={(e) => handleShiftChange(shift.id, 'startTime', e.target.value)}
                                className="border border-gray-300 rounded px-2 py-1.5 text-sm w-full focus:ring-2 focus:ring-blue-500 focus:border-blue-500"
                              />
                              <span className="text-slate-400">-</span>
                              <input 
                                type="time" 
                                value={shift.endTime}
                                onChange={(e) => handleShiftChange(shift.id, 'endTime', e.target.value)}
                                className="border border-gray-300 rounded px-2 py-1.5 text-sm w-full focus:ring-2 focus:ring-blue-500 focus:border-blue-500"
                              />
                            </div>
                          ) : (
                            <div className="text-sm font-medium text-slate-700 py-1.5 px-3 bg-gray-50 border border-gray-200 rounded text-center">
                              {shift.startTime} - {shift.endTime} (Fixed)
                            </div>
                          )}
                        </div>
                        <div className="space-y-1">
                          <label className="text-xs font-semibold text-slate-500">Hourly Rate ($)</label>
                          {canNegotiateRate ? (
                            <div className="relative">
                              <DollarSign size={14} className="absolute left-3 top-1/2 -translate-y-1/2 text-slate-400" />
                              <input 
                                type="number" 
                                min="0"
                                value={shift.rate}
                                onChange={(e) => handleShiftChange(shift.id, 'rate', e.target.value)}
                                className="pl-8 border border-gray-300 rounded px-3 py-1.5 text-sm w-full font-bold text-slate-800 focus:ring-2 focus:ring-blue-500 focus:border-blue-500"
                              />
                            </div>
                          ) : (
                            <div className="text-sm font-bold text-slate-500 py-1.5 px-3 bg-gray-100 border border-gray-200 rounded flex items-center justify-between cursor-not-allowed">
                              <span>${shift.rate} / hr</span>
                              <span className="text-[10px] uppercase">Locked</span>
                            </div>
                          )}
                        </div>
                      </div>
                    </div>
                  );
                })}
              </div>
            </div>
            {canNegotiateRate && (
               <div>
                  <h4 className="font-bold text-sm text-slate-700 uppercase tracking-wide mb-3 flex items-center gap-2">
                    <Plane size={16} /> Extras
                  </h4>
                  <label className="flex items-start gap-3 p-4 border border-gray-200 rounded-lg cursor-pointer hover:bg-slate-50 transition-colors">
                    <input 
                      type="checkbox" 
                      checked={formData.requestTravel}
                      onChange={(e) => setFormData(p => ({ ...p, requestTravel: e.target.checked }))}
                      className="mt-1 w-4 h-4 text-blue-600 rounded focus:ring-blue-500" 
                    />
                    <div>
                      <span className="block font-semibold text-sm text-slate-800">Request Travel Allowance</span>
                      <span className="block text-xs text-slate-500 mt-0.5">Ask the pharmacy to cover travel expenses for this placement.</span>
                    </div>
                  </label>
               </div>
            )}
          </form>
        </div>
        <div className="p-5 border-t border-gray-100 bg-gray-50 flex justify-end gap-3">
          <button type="button" onClick={onClose} className="px-5 py-2.5 text-sm font-semibold text-slate-600 hover:text-slate-800 hover:bg-slate-200 rounded-lg transition-colors">Cancel</button>
          <button type="submit" form="counter-form" className="px-6 py-2.5 text-sm font-bold text-white bg-blue-600 hover:bg-blue-700 rounded-lg shadow-sm active:scale-95 transition-all flex items-center gap-2">
            <MessageCircle size={16} /> Send Offer
          </button>
        </div>
      </div>
    </div>
  );
};

              <div className="bg-white p-3 rounded-lg border border-gray-200">

                 <p className="text-sm text-slate-800 flex items-start gap-2">
                   <MapPin size={16} className="text-slate-400 shrink-0 mt-0.5" /> 
                   {post.address}
                 </p>
              </div>

               <button 
                  onClick={() => setIsExpanded(false)}
                  className="mt-4 text-xs font-bold text-slate-500 hover:text-slate-800 flex items-center gap-1 md:hidden"
               >
                  <ChevronUp size={12}/> Close Details
               </button>
            </div>
          </div>
        </div>
      )}

      <button onClick={() => setIsExpanded(!isExpanded)} className="w-full bg-gray-50 hover:bg-gray-100 border-t border-gray-100 py-1.5 text-xs font-semibold text-slate-500 transition-colors hidden md:flex items-center justify-center gap-1">
        {isExpanded ? 'Hide Details' : 'Read More & Shift Breakdown'} {isExpanded ? <ChevronUp size={12}/> : <ChevronDown size={12}/>}
      </button>
    </div>
  );
};

// ... FilterSection Component (Same as before) ...
const FilterSection = ({ title, children, isOpen = true }) => {
  const [expanded, setExpanded] = useState(isOpen);
  return (
    <div className="mb-6 border-b border-gray-100 pb-6 last:border-0">
      <button 
        onClick={() => setExpanded(!expanded)}
        className="flex items-center justify-between w-full mb-3 group"
      >
        <h4 className="text-xs font-bold text-slate-500 uppercase tracking-wider group-hover:text-blue-600 transition-colors">{title}</h4>
        <ChevronDown size={14} className={`text-slate-400 transition-transform ${expanded ? 'rotate-180' : ''}`} />
      </button>
      {expanded && <div className="animate-in slide-in-from-top-2 duration-200">{children}</div>}
    </div>
  );
};

// --- MAIN APP COMPONENT ---

export default function App() {
  const [posts, setPosts] = useState(INITIAL_DATA);
  const [savedPostIds, setSavedPostIds] = useState(new Set());
  const [activeTab, setActiveTab] = useState('browse');
  
  // Updated Counter Post State to hold selection context
  const [activeCounterPost, setActiveCounterPost] = useState(null); // { post: object, selectedShifts: Set }
  const [expandedStates, setExpandedStates] = useState({});

  const [sortConfig, setSortConfig] = useState({ key: 'date', direction: 'ascending' });
  const [filterConfig, setFilterConfig] = useState({
    city: [],      
    roles: [],     
    minRate: 0,    
    search: '',
    timeOfDay: [],
    dateRange: { start: '', end: '' }, 
    onlyUrgent: false,
    negotiableOnly: false,
    flexibleOnly: false,
    travelProvided: false,
    accommodationProvided: false,
    bulkShiftsOnly: false
  });

  const [isSidebarOpen, setIsSidebarOpen] = useState(false);

  const locationGroups = useMemo(() => {
    const groups = {};
    INITIAL_DATA.forEach(post => {
      if (!groups[post.state]) groups[post.state] = new Set();
      groups[post.state].add(post.city);
    });
    return groups;
  }, []);

  const handleSortChange = (key) => {
    setSortConfig(prev => ({
      key,
      direction: prev.key === key && prev.direction === 'ascending' ? 'descending' : 'ascending'
    }));
  };

  const toggleFilter = (category, value) => {
    setFilterConfig(prev => {
      const current = prev[category];
      if (current.includes(value)) {
        return { ...prev, [category]: current.filter(item => item !== value) };
      } else {
        return { ...prev, [category]: [...current, value] };
      }
    });
  };

  const toggleBooleanFilter = (key) => {
    setFilterConfig(prev => ({ ...prev, [key]: !prev[key] }));
  };
  
  const toggleStateExpand = (state) => {
    setExpandedStates(prev => ({ ...prev, [state]: !prev[state] }));
  };

  const toggleStateSelection = (state, citiesInState) => {
    setFilterConfig(prev => {
      const currentCities = prev.city;
      const allSelected = citiesInState.every(c => currentCities.includes(c));
      let newCities;
      if (allSelected) {
        newCities = currentCities.filter(c => !citiesInState.includes(c));
      } else {
        const toAdd = citiesInState.filter(c => !currentCities.includes(c));
        newCities = [...currentCities, ...toAdd];
      }
      return { ...prev, city: newCities };
    });
  };

  const toggleSavePost = (id) => {
    setSavedPostIds(prev => {
      const newSet = new Set(prev);
      if (newSet.has(id)) newSet.delete(id);
      else newSet.add(id);
      return newSet;
    });
  };

  const clearAllFilters = () => {
    setFilterConfig({ 
      city: [], 
      roles: [], 
      minRate: 0, 
      search: '', 
      timeOfDay: [], 
      dateRange: { start: '', end: '' },
      onlyUrgent: false,
      negotiableOnly: false,
      flexibleOnly: false,
      travelProvided: false,
      accommodationProvided: false,
      bulkShiftsOnly: false
    });
  };

  const openCounterModal = (post, selectedShifts) => {
    setActiveCounterPost({ post, selectedShifts });
  };

  const handleCounterOfferSubmit = (data) => {
    console.log("Counter Offer Submitted:", data);
    setActiveCounterPost(null);
  };

  const processedPosts = useMemo(() => {
    let result = [...posts];
    if (activeTab === 'saved') {
      result = result.filter(p => savedPostIds.has(p.id));
    }
    result = result.filter(post => {
      const anyShiftMatches = (predicate) => post.shifts.some(predicate);
      if (filterConfig.onlyUrgent && !post.isUrgent) return false;
      if (filterConfig.negotiableOnly && !post.isNegotiable) return false;
      if (filterConfig.flexibleOnly && !post.isFlexible) return false;
      if (filterConfig.travelProvided && !post.hasTravel) return false;
      if (filterConfig.accommodationProvided && !post.hasAccommodation) return false;
      if (filterConfig.bulkShiftsOnly && post.shifts.length < 5) return false;
      if (filterConfig.city.length > 0 && !filterConfig.city.includes(post.city)) return false;
      if (filterConfig.roles.length > 0 && !filterConfig.roles.includes(post.role)) return false;
      if (filterConfig.timeOfDay.length > 0) {
        const matchesTime = anyShiftMatches(shift => {
          const hour = shift.startHour;
          const matchesMorning = filterConfig.timeOfDay.includes('morning') && hour < 12;
          const matchesAfternoon = filterConfig.timeOfDay.includes('afternoon') && hour >= 12 && hour < 17;
          const matchesEvening = filterConfig.timeOfDay.includes('evening') && hour >= 17;
          return matchesMorning || matchesAfternoon || matchesEvening;
        });
        if (!matchesTime) return false;
      }

      // NEW: Date Range Filter Logic
      // If a range is selected, show post if ANY shift overlaps with that range.
      if (filterConfig.dateRange.start || filterConfig.dateRange.end) {
        const startFilter = filterConfig.dateRange.start ? new Date(filterConfig.dateRange.start) : new Date('1970-01-01');
        const endFilter = filterConfig.dateRange.end ? new Date(filterConfig.dateRange.end) : new Date('2100-01-01');
        
        const hasOverlap = anyShiftMatches(shift => {
          const shiftDate = new Date(shift.date);
          return shiftDate >= startFilter && shiftDate <= endFilter;
        });
        if (!hasOverlap) return false;
      }

      const maxPostRate = Math.max(...post.shifts.map(s => s.rate));
      if (maxPostRate < filterConfig.minRate) return false;
      if (filterConfig.search) {
        const searchLower = filterConfig.search.toLowerCase();
        const matches = 
          post.pharmacy.toLowerCase().includes(searchLower) || 
          post.address.toLowerCase().includes(searchLower) ||
          post.role.toLowerCase().includes(searchLower);
        if (!matches) return false;
      }
      return true;
    });

    if (sortConfig.key) {
      result.sort((a, b) => {
        let aValue, bValue;
        if (sortConfig.key === 'date') {
          aValue = a.shifts[0].date;
          bValue = b.shifts[0].date;
        } else if (sortConfig.key === 'rate') {
          aValue = Math.max(...a.shifts.map(s => s.rate));
          bValue = Math.max(...b.shifts.map(s => s.rate));
        } else {
          aValue = a[sortConfig.key];
          bValue = b[sortConfig.key];
        }
        if (aValue < bValue) return sortConfig.direction === 'ascending' ? -1 : 1;
        if (aValue > bValue) return sortConfig.direction === 'ascending' ? 1 : -1;
        return 0;
      });
    }
    return result;
  }, [posts, filterConfig, sortConfig, activeTab, savedPostIds]);

  const uniqueRoles = [...new Set(INITIAL_DATA.map(s => s.role))];
  const activeFilterCount = 
    filterConfig.city.length + 
    filterConfig.roles.length + 
    filterConfig.timeOfDay.length + 
    (filterConfig.minRate > 0 ? 1 : 0) + 
    (filterConfig.search ? 1 : 0) +
    (filterConfig.onlyUrgent ? 1 : 0) +
    (filterConfig.negotiableOnly ? 1 : 0) +
    (filterConfig.flexibleOnly ? 1 : 0) +
    (filterConfig.travelProvided ? 1 : 0) +
    (filterConfig.accommodationProvided ? 1 : 0) + 
    (filterConfig.bulkShiftsOnly ? 1 : 0) +
    (filterConfig.dateRange.start || filterConfig.dateRange.end ? 1 : 0);

  return (
    <div className="min-h-screen bg-[#f7f8fa] font-sans text-slate-800">
      
      {activeCounterPost && (
        <CounterOfferModal 
          post={activeCounterPost.post}
          selectedShiftIds={activeCounterPost.selectedShifts}
          onClose={() => setActiveCounterPost(null)}
          onSubmit={handleCounterOfferSubmit}
        />
      )}

      {/* HEADER */}
      <header className="bg-white border-b border-gray-200 sticky top-0 z-30">
        <div className="max-w-7xl mx-auto px-4 h-16 flex items-center justify-between">
          <div className="flex items-center gap-2">
            <div className="bg-[#2f6fec] w-8 h-8 rounded-lg flex items-center justify-center text-white font-bold text-lg shadow-sm">C</div>
            <h1 className="font-bold text-xl tracking-tight text-[#0b1736] hidden sm:block">Chemist<span className="text-[#2f6fec]">Tasker</span></h1>
          </div>
          
          <div className="flex bg-slate-100 p-1 rounded-lg">
            <button onClick={() => setActiveTab('browse')} className={`px-4 py-1.5 text-sm font-semibold rounded-md transition-all ${activeTab === 'browse' ? 'bg-white text-slate-900 shadow-sm' : 'text-slate-500 hover:text-slate-700'}`}>Browse</button>
            <button onClick={() => setActiveTab('saved')} className={`px-4 py-1.5 text-sm font-semibold rounded-md transition-all flex items-center gap-2 ${activeTab === 'saved' ? 'bg-white text-slate-900 shadow-sm' : 'text-slate-500 hover:text-slate-700'}`}>
              Saved <span className="bg-slate-200 text-slate-600 text-[10px] px-1.5 py-0.5 rounded-full">{savedPostIds.size}</span>
            </button>
          </div>

          <button className="lg:hidden p-2 text-slate-600 hover:bg-slate-100 rounded-lg relative" onClick={() => setIsSidebarOpen(true)}>
            <SlidersHorizontal size={20} />
            {activeFilterCount > 0 && <span className="absolute top-1 right-1 w-2.5 h-2.5 bg-red-500 rounded-full border-2 border-white"></span>}
          </button>
        </div>
      </header>

      <div className="max-w-7xl mx-auto flex items-start gap-6 p-4 lg:p-6">
        
        <aside className={`fixed inset-0 z-40 bg-black/50 lg:hidden transition-opacity duration-300 ${isSidebarOpen ? 'opacity-100' : 'opacity-0 pointer-events-none'}`} onClick={() => setIsSidebarOpen(false)} />

        <div className={`fixed lg:sticky lg:top-24 inset-y-0 left-0 z-50 w-80 bg-white lg:bg-transparent lg:border-none border-r border-gray-200 shadow-xl lg:shadow-none transform transition-transform duration-300 ease-in-out lg:transform-none flex flex-col h-full lg:max-h-[calc(100vh-8rem)] overflow-hidden ${isSidebarOpen ? 'translate-x-0' : '-translate-x-full'}`}>
          <div className="flex items-center justify-between p-4 border-b border-gray-100 lg:hidden bg-white">
            <h2 className="font-bold text-lg">Filters</h2>
            <button onClick={() => setIsSidebarOpen(false)}><X size={20} /></button>
          </div>

          <div className="p-4 lg:p-0 overflow-y-auto bg-white lg:bg-transparent rounded-xl h-full scrollbar-hide">
            <div className="bg-white lg:border lg:border-gray-200 lg:shadow-sm lg:rounded-xl lg:p-5">
              
              {/* Search */}
              <div className="mb-6">
                <div className="relative">
                  <Search size={16} className="absolute left-3 top-3 text-slate-400" />
                  <input type="text" placeholder="Search pharmacy, role..." value={filterConfig.search} onChange={(e) => setFilterConfig(prev => ({ ...prev, search: e.target.value }))} className="w-full pl-9 pr-3 py-2.5 bg-slate-50 border border-gray-200 rounded-lg text-sm focus:outline-none focus:ring-2 focus:ring-blue-500/20 focus:border-blue-500 transition-all" />
                </div>
              </div>

              {/* Quick Toggles */}
              <div className="mb-6 flex gap-3">
                 <label className={`flex-1 cursor-pointer border rounded-lg p-3 flex flex-col items-center justify-center gap-2 transition-all ${filterConfig.onlyUrgent ? 'bg-orange-50 border-orange-200 text-orange-700' : 'bg-white border-gray-200 hover:border-gray-300 text-slate-600'}`}>
                    <input type="checkbox" className="hidden" checked={filterConfig.onlyUrgent} onChange={() => toggleBooleanFilter('onlyUrgent')} />
                    <Zap size={20} className={filterConfig.onlyUrgent ? "text-orange-500" : "text-slate-400"} />
                    <span className="text-xs font-bold">Urgent Only</span>
                 </label>
              </div>

              {/* Role Filter (MOVED TO TOP) */}
              <FilterSection title="Job Role">
                <div className="space-y-2">
                  {uniqueRoles.map(role => (
                    <label key={role} className="flex items-center gap-3 cursor-pointer group hover:bg-slate-50 p-1.5 -mx-1.5 rounded-md transition-colors">
                       <input 
                          type="checkbox" 
                          checked={filterConfig.roles.includes(role)}
                          onChange={() => toggleFilter('roles', role)}
                          className="w-4 h-4 rounded border-gray-300 text-blue-600 focus:ring-blue-500"
                        />
                      <span className="text-sm text-slate-600 group-hover:text-slate-900">{role}</span>
                    </label>
                  ))}
                </div>
              </FilterSection>

              {/* Date Range Filter (NEW) */}
              <FilterSection title="Date Range">
                <div className="space-y-3 relative">
                  {(filterConfig.dateRange.start || filterConfig.dateRange.end) && (
                    <button 
                      onClick={() => setFilterConfig(prev => ({...prev, dateRange: { start: '', end: '' }}))}
                      className="absolute -top-10 right-0 text-[10px] uppercase font-bold text-red-500 hover:text-red-700 flex items-center gap-1"
                    >
                      <RotateCcw size={10} /> Reset
                    </button>
                  )}
                  <div className="flex flex-col gap-1">
                    <label className="text-xs font-bold text-slate-500 uppercase">From</label>
                    <input 
                      type="date" 
                      value={filterConfig.dateRange.start}
                      onChange={(e) => setFilterConfig(p => ({ ...p, dateRange: { ...p.dateRange, start: e.target.value } }))}
                      className="w-full border border-gray-300 rounded px-2 py-1.5 text-sm focus:ring-2 focus:ring-blue-500 focus:border-blue-500"
                    />
                  </div>
                  <div className="flex flex-col gap-1">
                    <label className="text-xs font-bold text-slate-500 uppercase">To</label>
                    <input 
                      type="date" 
                      value={filterConfig.dateRange.end}
                      onChange={(e) => setFilterConfig(p => ({ ...p, dateRange: { ...p.dateRange, end: e.target.value } }))}
                      className="w-full border border-gray-300 rounded px-2 py-1.5 text-sm focus:ring-2 focus:ring-blue-500 focus:border-blue-500"
                    />
                  </div>
                </div>
              </FilterSection>

              {/* Perks & Benefits */}
              <FilterSection title="Perks & Benefits">
                 <div className="space-y-2">
                    <label className="flex items-center gap-3 cursor-pointer group p-1.5 -mx-1.5 hover:bg-slate-50 rounded transition-colors">
                      <input type="checkbox" checked={filterConfig.bulkShiftsOnly} onChange={() => toggleBooleanFilter('bulkShiftsOnly')} className="w-4 h-4 rounded border-gray-300 text-blue-600 focus:ring-blue-500" />
                      <div className="flex items-center gap-2 text-sm text-slate-600"><CalendarRange size={16} className="text-slate-400" /> Bulk Shifts (1 Week+)</div>
                    </label>
                    <label className="flex items-center gap-3 cursor-pointer group p-1.5 -mx-1.5 hover:bg-slate-50 rounded transition-colors">
                      <input type="checkbox" checked={filterConfig.negotiableOnly} onChange={() => toggleBooleanFilter('negotiableOnly')} className="w-4 h-4 rounded border-gray-300 text-blue-600 focus:ring-blue-500" />
                      <div className="flex items-center gap-2 text-sm text-slate-600"><HandCoins size={16} className="text-slate-400" /> Negotiable Rate</div>
                    </label>
                    <label className="flex items-center gap-3 cursor-pointer group p-1.5 -mx-1.5 hover:bg-slate-50 rounded transition-colors">
                      <input type="checkbox" checked={filterConfig.flexibleOnly} onChange={() => toggleBooleanFilter('flexibleOnly')} className="w-4 h-4 rounded border-gray-300 text-blue-600 focus:ring-blue-500" />
                      <div className="flex items-center gap-2 text-sm text-slate-600"><RefreshCw size={16} className="text-slate-400" /> Flexible Hours</div>
                    </label>
                    <label className="flex items-center gap-3 cursor-pointer group p-1.5 -mx-1.5 hover:bg-slate-50 rounded transition-colors">
                      <input type="checkbox" checked={filterConfig.travelProvided} onChange={() => toggleBooleanFilter('travelProvided')} className="w-4 h-4 rounded border-gray-300 text-blue-600 focus:ring-blue-500" />
                      <div className="flex items-center gap-2 text-sm text-slate-600"><Plane size={16} className="text-slate-400" /> Travel Paid</div>
                    </label>
                    <label className="flex items-center gap-3 cursor-pointer group p-1.5 -mx-1.5 hover:bg-slate-50 rounded transition-colors">
                      <input type="checkbox" checked={filterConfig.accommodationProvided} onChange={() => toggleBooleanFilter('accommodationProvided')} className="w-4 h-4 rounded border-gray-300 text-blue-600 focus:ring-blue-500" />
                      <div className="flex items-center gap-2 text-sm text-slate-600"><BedDouble size={16} className="text-slate-400" /> Accommodation</div>
                    </label>
                 </div>
              </FilterSection>

                <div className="space-y-4">

                    const cities = [...citiesSet].sort();
                    const isExpanded = !!expandedStates[state];
                    const allSelected = cities.every(c => filterConfig.city.includes(c));
                    const someSelected = cities.some(c => filterConfig.city.includes(c));
                    const isIndeterminate = someSelected && !allSelected;

                    return (
                      <div key={state} className="border-b border-gray-50 last:border-0 pb-2">
                        <div className="flex items-center justify-between py-2 group">
                           <label className="flex items-center gap-3 cursor-pointer flex-1">
                              <IndeterminateCheckbox 
                                checked={allSelected}
                                indeterminate={isIndeterminate}
                                onChange={() => toggleStateSelection(state, cities)}
                                className="w-4 h-4 rounded border-gray-300 text-blue-600 focus:ring-blue-500"
                              />
                              <span className="text-sm font-bold text-slate-700">{state}</span>
                           </label>
                           <button onClick={() => toggleStateExpand(state)} className="p-1 text-slate-400 hover:text-blue-600">
                              {isExpanded ? <ChevronUp size={16} /> : <ChevronDown size={16} />}
                           </button>
                        </div>
                        {isExpanded && (
                          <div className="pl-7 space-y-2 animate-in slide-in-from-top-1">
                             {cities.map(city => (
                               <label key={city} className="flex items-center gap-3 cursor-pointer group">
                                  <input 
                                      type="checkbox" 
                                      checked={filterConfig.city.includes(city)}
                                      onChange={() => toggleFilter('city', city)}
                                      className="w-3.5 h-3.5 rounded border-gray-300 text-blue-600 focus:ring-blue-500"
                                    />
                                  <span className="text-sm text-slate-600 group-hover:text-slate-900">{city}</span>
                               </label>
                             ))}
                          </div>
                        )}
                      </div>
                    );
                  })}
                </div>
              </FilterSection>

              {/* Time of Day */}
              <FilterSection title="Time of Day">
                 <div className="grid grid-cols-3 gap-2">
                    {[
                      { id: 'morning', label: 'Morning', icon: Sun },
                      { id: 'afternoon', label: 'Afternoon', icon: Sunset },
                      { id: 'evening', label: 'Evening', icon: Moon }
                    ].map(time => {
                      const isActive = filterConfig.timeOfDay.includes(time.id);
                      const Icon = time.icon;
                      return (
                        <button
                          key={time.id}
                          onClick={() => toggleFilter('timeOfDay', time.id)}
                          className={`flex flex-col items-center justify-center p-2 rounded-lg border text-xs font-medium transition-all ${isActive ? 'bg-blue-50 border-blue-200 text-blue-700' : 'bg-white border-gray-200 text-slate-500 hover:bg-gray-50'}`}
                        >
                          <Icon size={16} className="mb-1" />
                          {time.label}
                        </button>
                      )
                    })}
                 </div>
              </FilterSection>

              {/* Min Rate */}
              <FilterSection title={`Min Rate: $${filterConfig.minRate}/hr`}>
                <input 
                  type="range" min="0" max="100" step="5"
                  value={filterConfig.minRate}
                  onChange={(e) => setFilterConfig(p => ({ ...p, minRate: Number(e.target.value) }))}
                  className="w-full h-2 bg-slate-200 rounded-lg appearance-none cursor-pointer accent-blue-600"
                />
                <div className="flex justify-between text-xs text-slate-400 mt-2">
                  <span>Any</span>
                  <span>$100+</span>
                </div>
              </FilterSection>

              <div className="mt-4 pt-4 border-t border-gray-100 lg:hidden">
                <button onClick={() => setIsSidebarOpen(false)} className="w-full bg-[#2f6fec] text-white py-3 rounded-lg font-bold">
                  View Results
                </button>
              </div>
            </div>
          </div>
        </div>

        {/* MAIN LIST */}
        <main className="flex-1 min-w-0">
          <div className="flex flex-col sm:flex-row justify-between items-start sm:items-center gap-4 mb-6">
            <div>
              <h2 className="text-2xl font-bold text-[#0b1736]">{activeTab === 'saved' ? 'Saved Jobs' : 'Available Jobs'}</h2>
              <p className="text-slate-500 text-sm mt-1">
                {activeTab === 'saved' ? `${processedPosts.length} saved items` : `Showing ${processedPosts.length} opportunities`}
              </p>
            </div>

            <div className="relative group self-end sm:self-auto flex items-center gap-3">
              <span className="text-xs font-bold uppercase text-slate-400 tracking-wider hidden sm:inline">Sort:</span>
              <div className="flex rounded-lg bg-white border border-gray-200 p-0.5">
                <select 
                  className="appearance-none bg-transparent pl-3 pr-8 py-1.5 text-sm font-medium text-slate-700 cursor-pointer focus:outline-none"
                  value={sortConfig.key}
                  onChange={(e) => handleSortChange(e.target.value)}
                >
                  <option value="date">Date</option>
                  <option value="rate">Rate (High-Low)</option>
                  <option value="distance">Distance</option>
                </select>
                <div className="pointer-events-none absolute right-12 top-1/2 -translate-y-1/2 text-slate-400"><ChevronDown size={14} /></div>
                <button onClick={() => handleSortChange(sortConfig.key)} className="px-2 border-l border-gray-100 hover:text-blue-600 text-slate-500">
                   <ArrowUpDown size={14} className={sortConfig.direction === 'descending' ? 'rotate-180 transition-transform' : 'transition-transform'} />
                </button>
              </div>
            </div>
          </div>

          {/* Filters Bar */}
          {activeFilterCount > 0 && (
            <div className="flex flex-wrap items-center gap-2 mb-6">
              {filterConfig.onlyUrgent && (
                 <span className="inline-flex items-center gap-1 bg-orange-50 text-orange-700 border border-orange-100 px-3 py-1 rounded-full text-xs font-bold">
                  URGENT ONLY <button onClick={() => toggleBooleanFilter('onlyUrgent')}><X size={12} /></button>
                </span>
              )}
              {filterConfig.bulkShiftsOnly && (
                 <span className="inline-flex items-center gap-1 bg-teal-50 text-teal-700 border border-teal-100 px-3 py-1 rounded-full text-xs font-bold">
                  Bulk Shifts (1 Week+) <button onClick={() => toggleBooleanFilter('bulkShiftsOnly')}><X size={12} /></button>
                </span>
              )}
              {filterConfig.negotiableOnly && (
                 <span className="inline-flex items-center gap-1 bg-blue-50 text-blue-700 border border-blue-100 px-3 py-1 rounded-full text-xs font-bold">
                  Negotiable <button onClick={() => toggleBooleanFilter('negotiableOnly')}><X size={12} /></button>
                </span>
              )}
              {filterConfig.travelProvided && (
                 <span className="inline-flex items-center gap-1 bg-indigo-50 text-indigo-700 border border-indigo-100 px-3 py-1 rounded-full text-xs font-bold">
                  Travel <button onClick={() => toggleBooleanFilter('travelProvided')}><X size={12} /></button>
                </span>
              )}
              {filterConfig.accommodationProvided && (
                 <span className="inline-flex items-center gap-1 bg-purple-50 text-purple-700 border border-purple-100 px-3 py-1 rounded-full text-xs font-bold">
                  Accommodation <button onClick={() => toggleBooleanFilter('accommodationProvided')}><X size={12} /></button>
                </span>
              )}
              {filterConfig.roles.map(role => (
                <span key={role} className="inline-flex items-center gap-1 bg-white text-slate-700 border border-gray-200 px-3 py-1 rounded-full text-xs font-medium shadow-sm">
                  {role} <button onClick={() => toggleFilter('roles', role)} className="hover:text-red-500"><X size={12} /></button>
                </span>
              ))}
               {filterConfig.timeOfDay.map(time => (
                <span key={time} className="inline-flex items-center gap-1 bg-blue-50 text-blue-700 border border-blue-100 px-3 py-1 rounded-full text-xs font-medium">
                  {time} <button onClick={() => toggleFilter('timeOfDay', time)} className="hover:text-blue-900"><X size={12} /></button>
                </span>
              ))}
              <button onClick={clearAllFilters} className="text-xs font-bold text-red-500 hover:text-red-700 ml-auto">Clear All</button>
            </div>
          )}

          <div className="space-y-4 pb-20">
            {processedPosts.length > 0 ? (
              processedPosts.map(post => (
                <ShiftCard 
                  key={post.id} 
                  post={post} 
                  isSaved={savedPostIds.has(post.id)}
                  onToggleSave={toggleSavePost}
                  onCounterClick={openCounterModal}
                />
              ))
            ) : (
              <div className="text-center py-24 bg-white border border-gray-200 border-dashed rounded-xl">
                <div className="bg-slate-50 w-20 h-20 rounded-full flex items-center justify-center mx-auto mb-4">
                  <Filter size={32} className="text-slate-300" />
                </div>
                <h3 className="text-lg font-bold text-slate-800 mb-1">No jobs found</h3>
                <p className="text-slate-500 mb-6 max-w-xs mx-auto">We couldn't find any positions matching your specific criteria.</p>
                <button onClick={clearAllFilters} className="bg-slate-800 text-white px-6 py-2 rounded-lg text-sm font-semibold hover:bg-slate-700 transition-colors">
                  Reset Filters
                </button>
              </div>
            )}
          </div>
        </main>
      </div>
    </div>
  );
}
